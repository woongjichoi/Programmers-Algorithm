def solution(a, b):
    week={1:"FRI",2:"SAT",3:"SUN",4:"MON",5:"TUE",6:"WED",0:"THU"}
    month={1:0,2:1*31,3:1*31+1*29,4:2*31+1*29,5:2*31+1*29+1*30,6:3*31+1*29+1*30,7:3*31+1*29+2*30,8:4*31+1*29+2*30,9:5*31+1*29+2*30,10:5*31+1*29+3*30,11:6*31+1*29+3*30,12:6*31+1*29+4*30}
    days=month[a]+b
    return week[days%7]
